                            var number = new VcNumber({
                                $target: '.vc-number',
                                range: 2,
                                max: 88,
                                min: 1,
                                hasDecimal: true,
                                showRange: true
                            }).on('afterChange',function(e){
                                var $target = e.input;
                                alert($target.val());
                            });

                            number.render();
                    })
